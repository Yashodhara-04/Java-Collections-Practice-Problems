//3.  Create a TreeMap to store the mappings of words to their frequencies in a given text.

import java.util.TreeMap;
import java.util.Map;

public class HelloWorld{

     public static void main(String []args){
      
      String s = "Peter Piper picked a peck of pickled peppers.A peck of pickled peppers Peter Piper picked. If Peter Piper picked a peck of pickled peppers,Whereâ€™s the peck of pickled peppers Peter Piper picked";
      Map<String,Integer> treeSet = new TreeMap<>();
      String[] strArray = s.split("\\s+");
      for(String str : strArray)
      {
          treeSet.put(str, treeSet.getOrDefault(str, 0)+1);
      }
      System.out.println(treeSet);
     }
}

import java.util.TreeMap;

     }

class Student {
    private int age;
    private String name;
    private String grade;
    
    public Student(int age, String name, String grade)
    {
        this.age = age;
        this.name = name;
        this.grade = grade;
    }
    
    public int getAge()
    {
        return this.age;
    }
    
    public void setAge(int age)
    {
        this.age = age;
    }
    
    public String getName()
    {
        return this.name;
    }
    
    public void setName(String name)
    {
        this.name = name;
    }
    
    public String getGrade()
    {
        return this.grade;
    }
    
    public void setGrade(String grade)
    {
        this.grade = grade;
    }
    
   
    public String toString()
    {
        return "[ Emp Id = " + this.age + " || Name = " + this.name + " || Salary = " + this.grade + " ]";
    }
 }
